def find_2020():
    with open("../static/advent-2020-input-day-one.txt") as fi:
	numbers = list(map(float, fi.readlines()))

    for x in numbers:
	for y in numbers:
	    if x + y == 2020:
		return x * y

print(find_2020())

(defn is-2020? [v]
  (= 2020 (reduce + v)))


(defn find-2020 []
  (first
   (filter
    is-2020?
    (for [a x-input b x-input] [a b]))))
